def self.magic_method(klass, method, params='')
  ns = namespace(klass)
  my_name = klass.name.sub(ns, '').to_sym
  parent = (ns == '' ? Object : qualified_const_get(ns))
  unless parent.method_defined?(my_name)
    parent.class_eval <<-CLASS
      def #{my_name}(#{params})
        #{klass}.#{method}(#{params.gsub(/\s?=(.+?),/,',')})
      end
    CLASS
  end
end
